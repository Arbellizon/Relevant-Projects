use csc320_calvija

CREATE TABLE Customers (CustomerID Int NOT NULL AUTO_INCREMENT, Firstname VARCHAR(45) NOT NULL, 
	Lastname VARCHAR(45) NOT NULL, Email VARCHAR(100) NOT NULL, MailingAddress VARCHAR(100) NOT NULL,
	BillingAddress VARCHAR(100) NOT NULL, Phone VARCHAR(15) NOT NULL, PRIMARY KEY (CustomerID));


CREATE TABLE Cars (CarID INT NOT NULL AUTO_INCREMENT, RentalStartDate DATETIME NOT NULL,
 RentalEndDate DATETIME NOT NULL, PRIMARY KEY (CarID));

CREATE TABLE CarReservations (CarResID INT NOT NULL AUTO_INCREMENT, CarID INT NOT NULL,
 CustomerID INT NOT NULL, PRIMARY KEY (CarResID));


CREATE TABLE Flights (FlightID INT NOT NULL AUTO_INCREMENT, Destintation VARCHAR(3) NOT NULL, 
			DepartureDate DATETIME NOT NULL,ReturnDate DATETIME NOT NULL, PRIMARY KEY (FlightID));

CREATE TABLE FlightReservations (FlightRESID INT NOT NULL AUTO_INCREMENT, CustomerID INT NOT NULL, 
FlightID INT NOT NULL, PRIMARY KEY (FlightRESID));

CREATE TABLE Hotel (HotelID INT NOT NULL AUTO_INCREMENT, ArrivalDate DATETIME NOT NULL, 
DepartureDate DATETIME NOT NULL,PRIMARY KEY (HotelID));

CREATE TABLE HotelReservations ( HotelResId INT NOT NULL AUTO_INCREMENT, 
CustomerID INT NOT NULL, HotelID INT NOT NULL,PRIMARY KEY (HotelRESID));
