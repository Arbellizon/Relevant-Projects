	private boolean contains(Node x, String s, int d)

          char c = s.charAt(d);

          {
              
              return false;
	  
          }
          
	  if (c < x.c)
          {
              return contains(x.left, s, d);
          }
	  else if (c > x.c)
          {
              return contains(x.right, s, d);
          }
	  else if (d < s.length()-1) 
          {
              return contains(x.mid, s, d+1);
          }
	  else 
          {
              return x.end;
          }
